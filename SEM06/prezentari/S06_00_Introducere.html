Error Handling Robust

#!/bin/bash
# Activare mod strict
set -euo pipefail

# Exit codes standardizate
readonly EXIT_SUCCESS=0
readonly EXIT_FAILURE=1
readonly EXIT_INVALID_ARG=2
readonly EXIT_FILE_NOT_FOUND=3

# Trap pentru cleanup
cleanup() {
    local exit_code=$?
    rm -f "$TEMP_FILE" 2>/dev/null
    [[ $exit_code -ne 0 ]] && log_error "Script failed with code: $exit_code"
    exit "$exit_code"
}
trap cleanup EXIT

# Error handler cu stack trace
error_handler() {
    local line=$1 cmd=$2
    log_error "Error at line $line: $cmd"
    # Stack trace pentru debugging
    local i=0
    while caller $i; do ((i++)); done
}

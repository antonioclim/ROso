cleanup() {
    local exit_code=$?

trap cleanup EXIT ERR

# Error handler cu context
error_handler() {
    local line=$1 cmd=$2 code=$3
    log_error "Command '$cmd' failed at line $line (exit: $code)"
    # Stack trace
    local frame=0
    while caller $frame; do
        ((frame++))
    done
}

log() {
    local level=$1; shift
    local msg="$*"
    local timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    
    # Culori pentru terminal
    case $level in
        INFO)  color="\e[32m" ;;  # Verde
        WARN)  color="\e[33m" ;;  # Galben
        ERROR) color="\e[31m" ;;  # RoÈ™u
        *)     color="\e[0m"  ;;

        "$timestamp" "$level" "$msg" >&2
}

# Validare cu mesaje descriptive
validate_path() {
    local path=$1 type=$2
    
    [[ -z "$path" ]] && {
        log_error "Path cannot be empty"
        return 1
    }
    
    case $type in
        file)
            [[ -f "$path" ]] || {
                log_error "File not found: $path"
                return 1
            }
            [[ -r "$path" ]] || {
                log_error "File not readable: $path"
                return 1
            }
            ;;
        directory)
            [[ -d "$path" ]] || {
                log_error "Directory not found: $path"
                return 1
            }
            ;;
    esac
    return 0
}
